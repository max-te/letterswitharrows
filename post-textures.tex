\newcommand{\myarw@}[2]{
\setbox0\hbox{$\m@th#1#2$}%
\edef\mylen{\strip@pt\dimexpr((\wd0*10)/10)\relax}%
\vbox{
    \lineskiplimit=\maxdimen
    \baselineskip=0pt
    \lineskip=\dimexpr\f@size pt/10\relax
    \halign{##\cr
      $\m@th#1
      \mathchoice
            {\@rrr{\mylen}{1}{.25}}
            {\@rrr{\mylen}{\strip@pt\dimexpr\f@size pt/10\relax}{.25}}
            {\@rrr{\mylen}{.8}{.2}}
            {\@rrr{\mylen}{.6}{.2}}
      $%
      \cr
      \box0\cr
    }
}
}
\newcommand{\myarw}[1]{\mathchoice
    {\myarw@\displaystyle{#1}}%
    {\myarw@\textstyle{#1}}%
    {\myarw@\scriptstyle{#1}}%
    {\myarw@\scriptscriptstyle{#1}}
}
